\begin{frame}{Modèle Mathématique du PoC}
\begin{columns}
\begin{column}{0.47\textwidth}
\begin{block}{Équation d'évaporation}
$$Q_{evap} = k_{base} \times f(T) \times f(H)$$

\small
\begin{itemize}
    \item $k_{base} = 0.03$ mL/h
    \item $f(T) = 1 + 0.05(T - 37)$
    \item $f(H) = \frac{100-H}{100-95}$
\end{itemize}
\end{block}

\begin{exampleblock}{Exemple}
\small
À 70\% humidité :\\
$Q_{evap} = 0.03 \times 1 \times 6 = \textbf{0.18 mL/h}$\\
= \textcolor{danger}{6× plus rapide !}
\end{exampleblock}
\end{column}
\begin{column}{0.47\textwidth}
\begin{block}{Contrôleur PID}
$$Q_{pompe} = K_p \cdot e(t) + K_i \cdot \int e(\tau)d\tau$$

\small
\begin{itemize}
    \item $K_p = 0.5$ (proportionnel)
    \item $K_i = 0.1$ (intégral)
    \item $e(t) = V_{cible} - V_{mesure}$
\end{itemize}
\end{block}

\begin{block}{Saturation}
\small
$Q_{pompe} \in [0, 0.1]$ mL/min\\
(limites physiques de la pompe)
\end{block}
\end{column}
\end{columns}
\end{frame}
